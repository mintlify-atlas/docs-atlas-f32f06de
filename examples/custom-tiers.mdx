## Overview

Pricing tiers are defined in your `config/index.ts` file. Each tier follows the `PricingTier` interface and can be customized to fit your business model.

## Tier Structure

Here's the complete TypeScript interface for a pricing tier:

```tsx config/index.ts
export interface PricingTier {
  name: string;              // Display name of the tier
  id: string;                // Unique identifier
  price: Record<string, number | string>; // Prices for each frequency
  description: string;       // Short description
  features: string[];        // List of features
  cta: string;              // Call-to-action button text
  highlighted?: boolean;    // Show with highlighted styling
  popular?: boolean;        // Show "Most Popular" badge
}
```

## Example 1: Student/Education Tier

Create a special tier for students with discounted pricing:

<Steps>
  <Step title="Define the Tier">
    Add a student tier to your `TIERS` array:

    ```tsx config/index.ts
    export const TIERS: PricingTier[] = [
      {
        id: "student",
        name: "Student",
        price: {
          monthly: 5,
          yearly: 4,
        },
        description: "Perfect for students and educators",
        features: [
          "All Individual features",
          "50% student discount",
          "Educational resources",
          "Valid .edu email required",
          "5 projects",
        ],
        cta: "Verify Student Status",
        popular: true,
      },
      // ... other tiers
    ];
    ```
  </Step>

  <Step title="Handle Special CTA">
    If your student tier requires verification, update your click handler:

    ```tsx components/PricingCard.tsx
    <Button
      variant="expandIcon"
      Icon={ArrowRight}
      iconPlacement="right"
      onClick={() => {
        if (tier.id === "student") {
          // Redirect to student verification
          window.location.href = "/verify-student";
        } else {
          // Normal checkout flow
          handleCheckout(tier.id);
        }
      }}
    >
      {tier.cta}
    </Button>
    ```
  </Step>
</Steps>

## Multiple Pricing Frequencies

You can support different billing cycles:

```tsx config/index.ts
// Support quarterly billing
export const PAYMENT_FREQUENCIES = ["monthly", "quarterly", "yearly"];

export const TIERS: PricingTier[] = [
  {
    id: "pro",
    name: "Pro",
    price: {
      monthly: 29,
      quarterly: 25,  // ~14% discount
      yearly: 20,     // ~31% discount
    },
    description: "For professionals",
    features: [
      "All features",
      "Priority support",
      "Advanced analytics",
    ],
    cta: "Get Started",
  },
];
```

## Dynamic Tier Configuration

Load tiers from an API or CMS:

```tsx lib/pricing.ts
import { PricingTier } from "@/config";

export async function getPricingTiers(): Promise<PricingTier[]> {
  // Fetch from your API
  const response = await fetch("/api/pricing-tiers");
  const tiers = await response.json();
  
  return tiers;
}
```

Then use it in your component:

```tsx components/DynamicPricing.tsx
"use client";

import { PricingCard } from "@/components/PricingCard";
import { PricingHeader } from "@/components/PricingHeader";
import { PAYMENT_FREQUENCIES } from "@/config";
import { getPricingTiers } from "@/lib/pricing";
import { useState, useEffect } from "react";

export const DynamicPricing = () => {
  const [tiers, setTiers] = useState([]);
  const [selectedPaymentFreq, setSelectedPaymentFreq] = useState(
    PAYMENT_FREQUENCIES[0],
  );

  useEffect(() => {
    getPricingTiers().then(setTiers);
  }, []);

  return (
    <section className="flex flex-col items-center gap-10 py-10">
      <PricingHeader
        title="Plans and Pricing"
        subtitle="Choose the perfect plan for your needs"
        frequencies={PAYMENT_FREQUENCIES}
        selectedFrequency={selectedPaymentFreq}
        onFrequencyChange={setSelectedPaymentFreq}
      />

      <div className="grid w-full max-w-6xl gap-6 sm:grid-cols-2 xl:grid-cols-4">
        {tiers.map((tier, i) => (
          <PricingCard
            key={i}
            tier={tier}
            paymentFrequency={selectedPaymentFreq}
          />
        ))}
      </div>
    </section>
  );
};
```

## Best Practices

<AccordionGroup>
  <Accordion title="Keep features concise">
    Use short, clear feature descriptions. Aim for 3-5 words per feature.
    
    **Good:** "Priority email support"
    
    **Bad:** "Get prioritized email support with faster response times from our dedicated team"
  </Accordion>

  <Accordion title="Order tiers logically">
    Typically, order from least to most expensive, with free tiers first. Place your most popular tier in the center for visual balance.
  </Accordion>

  <Accordion title="Use popular and highlighted wisely">
    Only mark ONE tier as `popular: true` (your recommended option). Use `highlighted: true` sparingly for premium or enterprise tiers.
  </Accordion>

  <Accordion title="Consistent pricing structure">
    Every tier should have prices for all payment frequencies, even if they're the same value.
  </Accordion>
</AccordionGroup>
