## Overview

The shadcn-pricing-page components are built on top of Tailwind CSS and shadcn/ui, giving you full control over theming through CSS variables, Tailwind classes, and component customization.

## Dark Mode Support

The components automatically support dark mode through Tailwind's dark mode classes.

### Setting Up Dark Mode

<Steps>
  <Step title="Configure Tailwind">
    Ensure your `tailwind.config.ts` has dark mode enabled:

    ```ts tailwind.config.ts
    import type { Config } from "tailwindcss";

    const config: Config = {
      darkMode: ["class"], // or ["class", "selector", '[data-theme="dark"]']
      content: [
        "./pages/**/*.{ts,tsx}",
        "./components/**/*.{ts,tsx}",
        "./app/**/*.{ts,tsx}",
      ],
      // ... rest of config
    };

    export default config;
    ```
  </Step>

  <Step title="Add Dark Mode Toggle">
    Create a theme provider and toggle button:

    ```tsx components/ThemeProvider.tsx
    "use client";

    import { ThemeProvider as NextThemesProvider } from "next-themes";
    import { type ThemeProviderProps } from "next-themes/dist/types";

    export function ThemeProvider({ children, ...props }: ThemeProviderProps) {
      return <NextThemesProvider {...props}>{children}</NextThemesProvider>;
    }
    ```

    ```tsx app/layout.tsx
    import { ThemeProvider } from "@/components/ThemeProvider";

    export default function RootLayout({ children }) {
      return (
        <html lang="en" suppressHydrationWarning>
          <body>
            <ThemeProvider
              attribute="class"
              defaultTheme="system"
              enableSystem
              disableTransitionOnChange
            >
              {children}
            </ThemeProvider>
          </body>
        </html>
      );
    }
    ```
  </Step>

  <Step title="Test Dark Mode">
    The pricing components will automatically adapt to dark mode:

    ```tsx
    import { Pricing } from "@/components/Pricing";

    export default function PricingPage() {
      return <Pricing />;
    }
    ```
  </Step>
</Steps>

<Tip>
  The `PricingCard` component has special dark mode styling for highlighted and popular tiers, including different gradient overlays.
</Tip>

## Customizing Colors

```css app/globals.css

```

## Customizing the Popular Badge

The popular badge uses an orange color by default. You can customize it:

### Method 1: Inline Customization

Modify the `PricingCard.tsx` component:

```tsx components/PricingCard.tsx
{isPopular && (
  <Badge className="mt-1 bg-blue-600 px-1 py-0 text-white hover:bg-blue-700">
    ⭐ Recommended
  </Badge>
)}
```

### Method 2: CSS Variable

Define a custom variable and use it:

```css app/globals.css
@layer base {
  :root {
    --popular-badge: 217 91% 60%; /* Blue */
  }

  .dark {
    --popular-badge: 217 91% 70%;
  }
}
```

```tsx components/PricingCard.tsx
{isPopular && (
  <Badge className="mt-1 bg-[hsl(var(--popular-badge))] px-1 py-0 text-white">
    🔥 Most Popular
  </Badge>
)}
```

## Custom Card Layout

Create a completely custom card layout:

```tsx components/CustomPricingCard.tsx
import { TIERS } from "@/config";
import { Button } from "@/ui/button";
import { Check } from "lucide-react";

export const CustomPricingCard = ({
  tier,
  paymentFrequency,
}: {
  tier: (typeof TIERS)[0];
  paymentFrequency: string;
}) => {
  const price = tier.price[paymentFrequency];

  return (
    <div className="group relative overflow-hidden rounded-3xl border-2 border-gray-200 bg-white p-8 transition-all hover:border-primary hover:shadow-xl dark:border-gray-800 dark:bg-gray-900">
      {/* Header */}
      <div className="mb-6 text-center">
        <h3 className="text-2xl font-bold">{tier.name}</h3>
        <p className="mt-2 text-sm text-muted-foreground">
          {tier.description}
        </p>
      </div>

      {/* Price */}
      <div className="mb-8 text-center">
        <div className="text-5xl font-extrabold">
          {typeof price === "number" ? `$${price}` : price}
        </div>
        {typeof price === "number" && (
          <div className="text-sm text-muted-foreground">per user/month</div>
        )}
      </div>

      {/* Features */}
      <ul className="mb-8 space-y-3">
        {tier.features.map((feature, i) => (
          <li key={i} className="flex items-start gap-3">
            <Check className="mt-0.5 h-5 w-5 shrink-0 text-green-500" />
            <span className="text-sm">{feature}</span>
          </li>
        ))}
      </ul>

      {/* CTA */}
      <Button className="w-full" size="lg">
        {tier.cta}
      </Button>
    </div>
  );
};
```

## Animation Customization

Add custom animations and transitions:

```tsx components/PricingCard.tsx
import { motion } from "framer-motion";

export const AnimatedPricingCard = ({ tier, paymentFrequency, index }) => {
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ delay: index * 0.1 }}
      whileHover={{ scale: 1.05 }}
      className="relative flex flex-col gap-8 overflow-hidden rounded-2xl border p-6 shadow"
    >
      {/* Card content */}
    </motion.div>
  );
};
```

## Best Practices

<AccordionGroup>
  <Accordion title="Maintain contrast ratios">
    Ensure text remains readable in both light and dark modes. Use tools like [Contrast Checker](https://webaim.org/resources/contrastchecker/) to verify WCAG compliance.
  </Accordion>

  <Accordion title="Test on multiple devices">
    Verify your theming looks good on mobile, tablet, and desktop screens. Pay attention to how background effects render on different devices.
  </Accordion>

  <Accordion title="Keep animations subtle">
    Hover effects and transitions should enhance the experience, not distract. Respect `prefers-reduced-motion` for accessibility.
  </Accordion>

  <Accordion title="Use semantic color names">
    When defining custom colors, use semantic names like `--primary`, `--accent` rather than specific colors like `--blue-500`.
  </Accordion>
</AccordionGroup>
